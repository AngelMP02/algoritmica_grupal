Algoritmo clasificar4
    # Clasifica `a', `b' `c' y `d' en orden creciente.

Entrada
    a, b, c, d : T → COMPARABLE

precondición
    VERDADERO

realización
    clasificar3(a, b, c)
    # a ≤ b ≤ c ; situar `d'

    si
        d < a
    entonces
        # a > d ; b ≤ c
        intercambiar(a, d)
        # a ≤ d ; b ≤ c
    fin si
    # a ≤ d ; b ≤ c

    clasificar3(b, c , d)
    # a ≤ b ≤ c ≤ d

postcondición
    a ≤ b ≤ c ≤ d

fin clasificar4
